52 template <typename T>
53 inline error_code pq_assign_socket(T& conn) {
54  static_assert(Connection<T>, "T must be a Connection");
55  int fd = PQsocket(get_native_handle(conn));
56  if (fd == -1) {
57  return error::pq_socket_failed;
58  }
59 
60  int new_fd = dup(fd);
61  if (new_fd == -1) {
62  set_error_context(conn, "error while dup(fd) for socket stream");
63  return error_code{errno, boost::system::generic_category()};
64  }
65 
66  error_code ec;
67  get_socket(conn).assign(new_fd, ec);
68 
69  if (ec) {
70  set_error_context(conn, "assign socket failed");
71  }
72  return ec;
73 }
