108 template <typename Context, typename BinaryQuery>
109 struct async_send_query_params_op {
110  Context ctx_;
111  BinaryQuery query_;
112 
113  void perform() {
114  decltype(auto) conn = get_connection(ctx_);
115  if (auto ec = set_nonblocking(conn)) {
116  return done(ctx_, ec);
117  }
118  //In the nonblocking state, calls to PQsendQuery, PQputline,
119  //PQputnbytes, PQputCopyData, and PQendcopy will not block
120  //but instead return an error if they need to be called again.
121  while (!send_query_params(conn, query_));
122  post(ctx_, *this);
123  }
124 
125  void operator () (error_code ec = error_code{}, std::size_t = 0) {
126  // if data has been flushed or error has been set by
127  // read operation no write opertion handling is needed
128  // anymore.
129  if (get_query_state(ctx_) != query_state::send_in_progress) {
130  return;
131  }
132 
133  // In case of write operation error - finish the request
134  // with error.
135  if (ec) {
136  return done(ctx_, ec);
137  }
138 
139  // Trying to flush output one more time according to the
140  // documentation
141  switch (flush_output(get_connection(ctx_))) {
142  case query_state::error:
143  done(ctx_, error::pg_flush_failed);
144  break;
145  case query_state::send_in_progress:
146  write_poll(ctx_, *this);
147  break;
148  case query_state::send_finish:
149  set_query_state(ctx_, query_state::send_finish);
150  break;
151  }
152  }
153 
154  using executor_type = std::decay_t<decltype(impl::get_executor(ctx_))>;
155 
156  auto get_executor() const noexcept {
157  return impl::get_executor(ctx_);
158  }

189 template <typename Context, typename ResultProcessor>
190 struct async_get_result_op : boost::asio::coroutine {
191  Context ctx_;
192  ResultProcessor process_;
193 
194  async_get_result_op(Context ctx, ResultProcessor process)
195  : ctx_(ctx), process_(process) {}
196 
197  void perform() {
198  post(ctx_, *this);
199  }
200 
201  void done() {
202  return impl::done(ctx_);
203  }
204 
205  void done(error_code ec) {
206  set_error_context(get_connection(ctx_), "error while get request result");
207  return impl::done(ctx_, ec);
208  }
209 
210  void operator() (error_code ec = error_code{}, std::size_t = 0) {
211  // In case when query error state has been set by send query params
212  // operation skip handle and do nothing more.
213  if (get_query_state(ctx_) == query_state::error) {
214  return;
215  }
216 
217  if (ec) {
218  // Bad descriptor error can occur here if the connection
219  // has been closed by user during processing.
220  if (ec == asio::error::bad_descriptor) {
221  ec = asio::error::operation_aborted;
222  }
223  return done(ec);
224  }
225 
226  reenter(*this) {
227  while (is_busy(get_connection(ctx_))) {
228  yield read_poll(ctx_, *this);
229  if (auto err = consume_input(get_connection(ctx_))) {
230  return done(err);
231  }
232  }
233 
234  if (auto res = get_result(get_connection(ctx_))) {
235  const auto status = result_status(*res);
236  switch (status) {
237  case PGRES_SINGLE_TUPLE:
238  process_and_done(std::move(res));
239  return;
240  case PGRES_TUPLES_OK:
241  process_and_done(std::move(res));
242  consume_result(get_connection(ctx_));
243  return;
244  case PGRES_COMMAND_OK:
245  done();
246  consume_result(get_connection(ctx_));
247  return;
248  case PGRES_BAD_RESPONSE:
249  done(error::result_status_bad_response);
250  consume_result(get_connection(ctx_));
251  return;
252  case PGRES_EMPTY_QUERY:
253  done(error::result_status_empty_query);
254  consume_result(get_connection(ctx_));
255  return;
256  case PGRES_FATAL_ERROR:
257  done(result_error(*res));
258  consume_result(get_connection(ctx_));
259  return;
260  case PGRES_COPY_OUT:
261  case PGRES_COPY_IN:
262  case PGRES_COPY_BOTH:
263  case PGRES_NONFATAL_ERROR:
264  break;
265  }
266  set_error_context(get_connection(ctx_), get_result_status_name(status));
267  done(error::result_status_unexpected);
268  consume_result(get_connection(ctx_));
269  } else {
270  done();
271  }
272  }
273  }
