78 template <class ElementsT>
79 struct query_builder {
80  using elements_type = ElementsT;
81 
82  elements_type elements;
83 
84  constexpr auto text() const noexcept {
85  using namespace detail;
86  using namespace hana::literals;
87  return hana::fold(elements, hana::tuple<hana::size_t<1>, nothing_t>{},
88  [] (auto r, const auto& e) {
89  using tag_type = typename std::decay_t<decltype(e)>::tag_type;
90  if constexpr (std::is_same_v<tag_type, query_text_tag>) {
91  return hana::make_tuple(r[0_c], r[1_c] + std::move(e.value));
92  } else if constexpr (std::is_same_v<tag_type, query_param_tag>) {
93  return hana::make_tuple(incremented(r[0_c]), detail::concat(r[1_c], "$"_s + to_string(r[0_c])));
94  } else {
95  static_assert(std::is_same_v<tag_type, void>, "unsupported tag type");
96  }
97  })[1_c];
98  }
99 
100  constexpr auto params() const noexcept {
101  using namespace detail;
102  return hana::fold(elements, hana::tuple<>(),
103  [] (auto&& r, auto&& e) {
104  using tag_type = typename std::decay_t<decltype(e)>::tag_type;
105  if constexpr (std::is_same_v<tag_type, query_text_tag>) {
106  return r;
107  } else if constexpr (std::is_same_v<tag_type, query_param_tag>) {
108  return hana::append(std::move(r), std::move(e.value));
109  } else {
110  static_assert(std::is_same_v<tag_type, void>, "unsupported tag type");
111  }
112  });
113  }
114 
115  constexpr auto build() const {
116  return hana::unpack(
117  params(),
118  [&] (auto&& ... params) {
119  return make_query(text(), std::move(params) ...);
120  }
121  );
122  }
