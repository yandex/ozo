80 template <typename Context>
81 struct async_connect_op {
82  Context context;
83 
84  void done(error_code ec = error_code {}) {
85  get_timer(get_connection(context)).cancel();
86 
87  get_io_context(get_connection(context))
88  .post(detail::bind(std::move(get_handler(context)), std::move(ec), std::move(get_connection(context))));
89  }
90 
91  void perform(const std::string& conninfo, const time_traits::duration& timeout) {
92  if (error_code ec = start_connection(get_connection(context), conninfo)) {
93  return done(ec);
94  }
95 
96  if (connection_bad(get_connection(context))) {
97  return done(error::pq_connection_status_bad);
98  }
99 
100  if (error_code ec = assign_socket(get_connection(context))) {
101  return done(ec);
102  }
103 
104  get_timer(get_connection(context)).expires_after(timeout);
105  get_timer(get_connection(context)).async_wait(bind_executor(get_executor(context),
106  make_timeout_handler(get_socket(get_connection(context)))));
107 
108  return write_poll(get_connection(context), bind_executor(get_executor(context), *this));
109  }
110 
111  void operator () (error_code ec, std::size_t = 0) {
112  if (ec) {
113  set_error_context(get_connection(context), "error while connection polling");
114  return done(ec);
115  }
116 
117  switch (connect_poll(get_connection(context))) {
118  case PGRES_POLLING_OK:
119  return done();
120 
121  case PGRES_POLLING_WRITING:
122  return write_poll(get_connection(context), std::move(*this));
123 
124  case PGRES_POLLING_READING:
125  return read_poll(get_connection(context), std::move(*this));
126 
127  case PGRES_POLLING_FAILED:
128  case PGRES_POLLING_ACTIVE:
129  break;
130  }
131 
132  done(error::pq_connect_poll_failed);
133  }
