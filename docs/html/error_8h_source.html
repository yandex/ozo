405 
406  std::string message(int value) const override {
407  switch (code(value)) {
408  case ok:
409  return "no error";
410  case pq_connection_start_failed:
411  return "pq_connection_start_failed - PQConnectionStart function failed";
412  case pq_socket_failed:
413  return "pq_socket_failed - PQSocket returned -1 as fd - it seems like there is no connection";
414  case pq_connection_status_bad:
415  return "pq_connection_status_bad - PQstatus returned CONNECTION_BAD";
416  case pq_connect_poll_failed:
417  return "pq_connect_poll_failed - PQconnectPoll function failed";
418  case oid_type_mismatch:
419  return "no conversion possible from oid to user-supplied type";
420  case unexpected_eof:
421  return "unexpected EOF while data read";
422  case pg_send_query_params_failed:
423  return "pg_send_query_params_failed - PQsendQueryParams function failed";
424  case pg_consume_input_failed:
425  return "pg_consume_input_failed - PQconsumeInput function failed";
426  case pg_set_nonblocking_failed:
427  return "pg_set_nonblocking_failed - PQsetnonblocking function failed";
428  case pg_flush_failed:
429  return "pg_flush_failed - PQflush function failed";
430  case bad_result_process:
431  return "bad_result_process - error while processing or converting result from the database";
432  case no_sql_state_found:
433  return "no_sql_state_found - no sql state has been found in database query execution error reply";
434  case result_status_unexpected:
435  return "result_status_unexpected - got unexpected status from query result";
436  case result_status_empty_query:
437  return "result_status_empty_query - the string sent to the server was empty";
438  case result_status_bad_response:
439  return "result_status_bad_response - the server's response was not understood";
440  case oid_request_failed:
441  return "error during request oids from a database";
442  }
443  return "no message for value: " + std::to_string(value);
444  }
