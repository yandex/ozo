81  void perform() {
82  decltype(auto) conn = get_connection(ctx_);
83  if (auto ec = set_nonblocking(conn)) {
84  return done(ctx_, ec);
85  }
86 
87  if (!send_query_params(conn, query_)) {

89  }
90

94  void operator () (error_code ec = error_code{}, std::size_t = 0) {
95  // if data has been flushed or error has been set by
96  // read operation no write operation handling is needed
97  // anymore.
98  if (get_query_state(ctx_) != query_state::send_in_progress) {
99  return;
100  }
101 
102  // In case of write operation error - finish the request
103  // with error.
104  if (ec) {
105  return done(ctx_, ec);
106  }
107 
108  // Trying to flush output one more time according to the
109  // documentation
110  switch (flush_output(get_connection(ctx_))) {
111  case query_state::error:
112  done(ctx_, error::pg_flush_failed);
113  break;
114  case query_state::send_in_progress:
115  get_connection(ctx_).async_wait_write(std::move(*this));
116  break;
117  case query_state::send_finish:
118  set_query_state(ctx_, query_state::send_finish);
119  break;
120  }
121  }

152 struct async_get_result_op : boost::asio::coroutine {

162  (*this)();
163  }

166  return impl::done(ctx_);

169  void done(error_code ec) {
170  if (std::empty(get_error_context(get_connection(ctx_)))) {
171  get_connection(ctx_).set_error_context("error while get request result");
172  }
173  return impl::done(ctx_, ec);
174  }
175 
176  void operator() (error_code ec = error_code{}, std::size_t = 0) {
177  // In case when query error state has been set by send query params
178  // operation skip handle and do nothing more.
179  if (get_query_state(ctx_) == query_state::error) {
180  return;
181  }
182 
183  if (ec) {
184  // Bad descriptor error can occur here if the connection
185  // has been closed by user during processing.
186  if (ec == asio::error::bad_descriptor) {
187  ec = asio::error::operation_aborted;
188  }
189  return done(ec);
190  }
191 
192  reenter(*this) {
193  while (is_busy(get_connection(ctx_))) {
194  yield get_connection(ctx_).async_wait_read(std::move(*this));
195  if (auto err = consume_input(get_connection(ctx_))) {
196  return done(err);
197  }
198  }
199 
200  result_ = get_result(get_connection(ctx_));
201 
202  if (!result_) {
203  return done();
204  }
205 
206  if (result_status(*result_) != PGRES_SINGLE_TUPLE) {
207  do {
208  while (is_busy(get_connection(ctx_))) {
209  yield get_connection(ctx_).async_wait_read(std::move(*this));
210  if (consume_input(get_connection(ctx_))) {
211  return handle_result();
212  }
213  }
214  } while (get_result(get_connection(ctx_)));
215  }
216 
217  handle_result();
218  }
219  }
220 
221  void handle_result() {
222  const auto status = result_status(*result_);
223  switch (status) {
224  case PGRES_SINGLE_TUPLE:
225  case PGRES_TUPLES_OK:
226  case PGRES_COMMAND_OK:
227  process_and_done(std::move(result_));
228  return;
229  case PGRES_BAD_RESPONSE:
230  done(error::result_status_bad_response);
231  return;
232  case PGRES_EMPTY_QUERY:
233  done(error::result_status_empty_query);
234  return;
235  case PGRES_FATAL_ERROR:
236  done(result_error(*result_));
237  return;
238  case PGRES_COPY_OUT:
239  case PGRES_COPY_IN:
240  case PGRES_COPY_BOTH:
241  case PGRES_NONFATAL_ERROR:
242  break;
243  }
244 
245  get_connection(ctx_).set_error_context(get_result_status_name(status));
246  done(error::result_status_unexpected);
247  }
